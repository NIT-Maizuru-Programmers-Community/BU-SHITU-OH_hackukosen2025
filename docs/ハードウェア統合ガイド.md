## システム構成概要

```
┌─────────────────────────────────────────────────────────────┐
│                       ☆ 部室王 ☆                            │
│                    Next.js Web App                           │
│                   (Vercel + Firebase)                        │
└─────────────────┬───────────────────────┬───────────────────┘
                  │                       │
         ┌────────▼────────┐     ┌───────▼────────┐
         │  NFCリーダー     │     │  ディスプレイ   │
         │  (部室入口)      │     │  (部室内設置)   │
         └─────────────────┘     └────────────────┘
```

### 通信フロー

1. **NFC → Web App**: IC カードをタッチ → API にカード ID 送信 → ログイン処理
2. **Web App → Display**: リアルタイムデータを Firestore 経由で表示に反映

#### 2.2 NFC 読み取りスクリプト

以下のスクリプトを Raspberry Pi に配置してください。

**`nfc_reader.py`**

```python
#!/usr/bin/env python3
"""
NFC カードリーダースクリプト
部室王 - NFCログインシステム
"""

import nfc
import requests
import time
import os
from dotenv import load_dotenv
from datetime import datetime

# 環境変数読み込み
load_dotenv()

# 設定
API_ENDPOINT = os.getenv('API_ENDPOINT', 'https://your-app.vercel.app/api/attendance/checkin')
API_KEY = os.getenv('API_KEY', '')
POLL_INTERVAL = 1  # カード読み取りの間隔（秒）

def on_connect(tag):
    """
    NFC カードがタッチされた時の処理
    """
    print(f"[{datetime.now()}] カード検出!")

    # カード ID を取得
    card_id = tag.identifier.hex()
    print(f"カード ID: {card_id}")

    # Web API にチェックインリクエストを送信
    try:
        response = requests.post(
            API_ENDPOINT,
            json={
                'nfcCardId': card_id,
                'timestamp': datetime.now().isoformat()
            },
            headers={
                'Content-Type': 'application/json',
                'X-API-Key': API_KEY
            },
            timeout=10
        )

        if response.status_code == 200:
            data = response.json()
            print(f"✓ ログイン成功: {data.get('displayName', '不明')}")
            print(f"  ポイント: {data.get('points', 0)}pt")

            # 成功時の音/LED 制御 (オプション)
            # play_success_sound()

        elif response.status_code == 404:
            print("✗ 未登録のカードです")
            # play_error_sound()

        else:
            print(f"✗ エラー: {response.status_code}")
            print(response.text)

    except requests.exceptions.RequestException as e:
        print(f"✗ 通信エラー: {e}")

    # カードが離れるまで待機
    while tag.is_present:
        time.sleep(0.1)

    print("カードが離れました\n")
    return True

def main():
    """
    メイン処理
    """
    print("="*50)
    print("  ☆ 部室王 ☆ NFC ログインシステム")
    print("="*50)
    print(f"API エンドポイント: {API_ENDPOINT}")
    print("カードをかざしてください...\n")

    # NFC リーダーに接続
    clf = nfc.ContactlessFrontend('usb')

    try:
        while True:
            # カードの読み取りを待機
            clf.connect(rdwr={
                'on-connect': on_connect,
                'beep-on-connect': False
            })
            time.sleep(POLL_INTERVAL)

    except KeyboardInterrupt:
        print("\n終了します...")
    finally:
        clf.close()

if __name__ == '__main__':
    main()
```

### 4. Web App 側の API 実装

#### 4.1 チェックイン API

**`app/api/attendance/checkin/route.ts`**

```typescript
import { NextRequest, NextResponse } from "next/server";
import { adminDb } from "@/lib/firebase-admin";
import { FieldValue } from "firebase-admin/firestore";

// API キー検証 (環境変数に設定)
const API_KEY = process.env.NFC_API_KEY;

export async function POST(request: NextRequest) {
	try {
		// API キー検証
		const apiKey = request.headers.get("X-API-Key");
		if (apiKey !== API_KEY) {
			return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
		}

		const { nfcCardId, timestamp } = await request.json();

		if (!nfcCardId) {
			return NextResponse.json(
				{ error: "nfcCardId is required" },
				{ status: 400 }
			);
		}

		// NFC カード ID でユーザーを検索
		const usersRef = adminDb.collection("users");
		const userSnapshot = await usersRef
			.where("nfcCardId", "==", nfcCardId)
			.limit(1)
			.get();

		if (userSnapshot.empty) {
			return NextResponse.json(
				{ error: "NFC card not registered" },
				{ status: 404 }
			);
		}

		const userDoc = userSnapshot.docs[0];
		const userId = userDoc.id;
		const userData = userDoc.data();

		// チェックイン処理
		const now = new Date();
		const today = now.toISOString().split("T")[0];

		// 1. 在室状態を更新
		await usersRef.doc(userId).update({
			isInRoom: true,
			lastCheckIn: FieldValue.serverTimestamp(),
			updatedAt: FieldValue.serverTimestamp(),
		});

		// 2. attendance コレクションに記録
		await adminDb.collection("attendance").add({
			userId,
			displayName: userData.displayName,
			status: "in",
			checkInTime: FieldValue.serverTimestamp(),
			updatedAt: FieldValue.serverTimestamp(),
		});

		// 3. ログインボーナス処理 (1日1回)
		const lastBonusDate = userData.lastBonusDate
			?.toDate()
			.toISOString()
			.split("T")[0];
		let bonusAwarded = false;

		if (lastBonusDate !== today) {
			const DAILY_BONUS = 10; // 1日のログインボーナス

			await usersRef.doc(userId).update({
				totalPoints: FieldValue.increment(DAILY_BONUS),
				lastBonusDate: FieldValue.serverTimestamp(),
			});

			// ポイント履歴に記録
			await adminDb.collection("pointHistory").add({
				userId,
				type: "login_bonus",
				amount: DAILY_BONUS,
				description: "ログインボーナス",
				timestamp: FieldValue.serverTimestamp(),
			});

			bonusAwarded = true;
		}

		// レスポンス
		return NextResponse.json({
			success: true,
			userId,
			displayName: userData.displayName,
			points: userData.totalPoints + (bonusAwarded ? 10 : 0),
			bonusAwarded,
			timestamp: now.toISOString(),
		});
	} catch (error) {
		console.error("Checkin error:", error);
		return NextResponse.json(
			{ error: "Internal server error" },
			{ status: 500 }
		);
	}
}
```

#### 2.3 ディスプレイ専用ページの作成

**`app/display/page.tsx`**

```typescript
"use client";

import { useEffect, useState } from "react";
import {
	collection,
	query,
	orderBy,
	limit,
	onSnapshot,
} from "firebase/firestore";
import { db } from "@/lib/firebase";

export default function DisplayPage() {
	const [ranking, setRanking] = useState([]);
	const [attendance, setAttendance] = useState([]);
	const [raceResult, setRaceResult] = useState(null);

	useEffect(() => {
		// ランキングのリアルタイム更新
		const rankingQuery = query(
			collection(db, "users"),
			orderBy("totalPoints", "desc"),
			limit(5)
		);
		const unsubscribeRanking = onSnapshot(rankingQuery, (snapshot) => {
			const data = snapshot.docs.map((doc) => ({ id: doc.id, ...doc.data() }));
			setRanking(data);
		});

		// 在室状況のリアルタイム更新
		const attendanceQuery = query(
			collection(db, "users"),
			where("isInRoom", "==", true)
		);
		const unsubscribeAttendance = onSnapshot(attendanceQuery, (snapshot) => {
			const data = snapshot.docs.map((doc) => ({ id: doc.id, ...doc.data() }));
			setAttendance(data);
		});

		// レース結果の取得 (最新)
		const raceQuery = query(
			collection(db, "races"),
			orderBy("raceDate", "desc"),
			limit(1)
		);
		const unsubscribeRace = onSnapshot(raceQuery, (snapshot) => {
			if (!snapshot.empty) {
				setRaceResult({ id: snapshot.docs[0].id, ...snapshot.docs[0].data() });
			}
		});

		return () => {
			unsubscribeRanking();
			unsubscribeAttendance();
			unsubscribeRace();
		};
	}, []);

	return (
		<div className='h-screen w-screen bg-gradient-to-br from-purple-900 via-blue-900 to-indigo-900 text-white p-8'>
			{/* ヘッダー */}
			<header className='text-center mb-8'>
				<h1 className='text-6xl font-bold mb-2'>☆ 部室王 ☆</h1>
				<p className='text-2xl opacity-80'>舞鶴高専 プロコン部</p>
			</header>

			<div className='grid grid-cols-2 gap-8 h-[calc(100vh-200px)]'>
				{/* 左: ランキング */}
				<div className='bg-white/10 backdrop-blur rounded-3xl p-6'>
					<h2 className='text-4xl font-bold mb-6'>ランキング</h2>
					<div className='space-y-4'>
						{ranking.map((user, index) => (
							<div
								key={user.id}
								className='flex items-center gap-4 bg-white/5 rounded-xl p-4'
							>
								<div className='text-5xl font-bold w-16 text-center'>
									{index + 1}
								</div>
								<div className='flex-1'>
									<div className='text-2xl font-semibold'>
										{user.displayName}
									</div>
									<div className='text-xl opacity-70'>
										{user.totalPoints} pt
									</div>
								</div>
							</div>
						))}
					</div>
				</div>

				{/* 右: 在室状況 & レース結果 */}
				<div className='space-y-8'>
					{/* 在室状況 */}
					<div className='bg-white/10 backdrop-blur rounded-3xl p-6'>
						<h2 className='text-4xl font-bold mb-6'>
							在室中 ({attendance.length}人)
						</h2>
						<div className='grid grid-cols-3 gap-4'>
							{attendance.map((user) => (
								<div
									key={user.id}
									className='bg-white/5 rounded-xl p-4 text-center'
								>
									<div className='text-xl font-semibold'>
										{user.displayName}
									</div>
								</div>
							))}
						</div>
					</div>

					{/* レース結果 */}
					{raceResult && (
						<div className='bg-white/10 backdrop-blur rounded-3xl p-6'>
							<h2 className='text-4xl font-bold mb-6'>本日のレース</h2>
							{raceResult.status === "completed" ? (
								<div>
									<div className='text-3xl mb-4'>
										優勝: {raceResult.winner?.displayName}
									</div>
									{/* レース詳細表示 */}
								</div>
							) : (
								<div className='text-2xl opacity-70'>レース開催前</div>
							)}
						</div>
					)}
				</div>
			</div>
		</div>
	);
}
```

## セキュリティ考慮事項

### 1. API キーの管理

- **環境変数に保存**: `.env` ファイル (Git にコミットしない)
- **定期的なローテーション**: API キーを定期的に変更
- **異なるキーを使用**: 本番環境と開発環境で異なるキーを設定

### 2. カード ID の暗号化

NFC カード ID をそのまま保存せず、ハッシュ化することを推奨:

```typescript
import crypto from "crypto";

function hashCardId(cardId: string): string {
	return crypto
		.createHash("sha256")
		.update(cardId + process.env.CARD_ID_SALT)
		.digest("hex");
}
```

### 3. レート制限

同一カードでの連続ログインを制限:

```typescript
// 最後のチェックインから5分以内は無視
const lastCheckIn = userData.lastCheckIn?.toDate();
if (lastCheckIn && now - lastCheckIn < 5 * 60 * 1000) {
	return NextResponse.json({ error: "Too soon" }, { status: 429 });
}
```
